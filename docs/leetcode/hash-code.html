var isAnagram = function (s, t) {
  if (s.length !== t.length) return false

  const record = new Array(26).fill(0)

  const firstCharCode = 'a'.charCodeAt()
  for (let i of s) {
    record[i.charCodeAt() - firstCharCode]++
  }
  for (let i of t) {
    record[i.charCodeAt() - firstCharCode]--
  }

  return record.every((r) => !r)
}
