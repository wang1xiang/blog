var dailyTemperatures = function (temperatures) {
  const n = temperatures.length
  const result = new Array(n).fill(0)

  const stack = [0]

  for (let i = 1; i < n; i++) {
    while (
      stack.length &&
      temperatures[i] > temperatures[stack[stack.length - 1]]
    ) {
      const top = stack.pop()
      result[top] = i - top
    }

    stack.push(i)
  }

  return result
}
