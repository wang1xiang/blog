定义react ref子组件

// 子组件
export interface RefFunType {
  changeText: (data) => void;
  replaceCaret: () => void;
}
// 暴露事件
useImperativeHandle(ref, () => ({
  changeText,
  replaceCaret,
}));

// 父组件
const refEdit = useRef<RefFunType>();
