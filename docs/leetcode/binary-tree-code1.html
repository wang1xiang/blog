    const temp = right
    right = left
    left = temp
    root.left = left
    root.right = right

      invertNode(node, node.left, node.right)

function minDepth(root) {
  if (root === null) return 0
  let queue = [root]
  let depth = 0
  while (queue.length) {
    let length = queue.length
    depth++
