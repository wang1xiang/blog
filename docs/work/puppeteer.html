      if (status !== 'Pending' &amp;&amp; status !== 'In progress') {
        clearInterval(timer)
        process.exit(0)
      }
      prevStatus = status
    }, 400)
  } catch (e) {
    console.log(chalk.red(e.message))
    if (e.message.includes('Cannot find module \'puppeteer\'')) {
      console.log(chalk.red('è¯·æ›´æ–°ä¾èµ–'))
      process.exit(1)
    } else {
      if (e.message.includes(branchChangedErrorMessage)) {
        console.log(chalk.yellow('åˆ†æ”¯å·²æ›´æ–°ï¼Œæ­£åœ¨é‡æ–°åŠ è½½'))
      } else {
        console.log(chalk.yellow('æœªçŸ¥é”™è¯¯ï¼Œæ­£åœ¨å°è¯•é‡æ–°å¯åŠ¨'))
      }
      if (browser) {
        await browser.close()
      }
      autoBuild()
    }
  }

async function handleBuildStatus(page, buildId, prevStatus) {
  // å‡ ç§æƒ…å†µ æ’é˜Ÿä¸­ æ„å»ºä¸­ å·²å–æ¶ˆ å·²å¤±è´¥ å·²æˆåŠŸ
  let status
  try {
    const attr = await page.$eval(`a.build-status-link[href*='${buildId}']`, e => e.getAttribute('tooltip'))
    if (attr) {
      status = ['Success', 'In progress', 'Failed', 'Aborted'].find(status => attr.startsWith(status))
      if (status === 'Aborted') {
        oneLineLog.clear()
        oneLineLog('')
        console.log(chalk.gray('æ„å»ºè¢«å–æ¶ˆï¼'))
      } else if (status === 'Failed') {
        oneLineLog.clear()
        oneLineLog('')
        console.log(chalk.red('æ„å»ºå¤±è´¥ï¼'))
      } else if (status === 'Success') {
        oneLineLog.clear()
        oneLineLog('')
        console.log(chalk.green('æ„å»ºæˆåŠŸï¼'))
      } else if (status === 'In progress') {
        if (prevStatus !== status) {
          oneLineLog.clear()
          oneLineLog('')
          console.log(chalk.cyanBright('å¼€å§‹æ„å»º:'))
        }
        const width = await page.$eval(`.progress-bar[href*='${buildId}'] .progress-bar-done`, e => e.style.width)
        oneLineLog(chalk.blue(`ä»»åŠ¡${buildId}: æ„å»ºè¿›åº¦${width}`))
      }
    }
  } catch (e) {
    // æ²¡æ‰¾åˆ°æ˜¯pending
    status =  'Pending'
    const pendingList = await page.$$('.svg-icon.icon-nobuilt, .svg-icon[class*="-anime"]')
    const pendingLength = pendingList.length > 1 ? pendingList.length - 1 : 0
    if (pendingLength) {
      if (pendingTipDotCount < maxPendingTipDotCount) {
        pendingTipDotCount += 1
      } else {
        pendingTipDotCount = 0
      }
      oneLineLog(chalk.blue(`æ’é˜Ÿä¸­ï¼Œå‰é¢è¿˜æœ‰${pendingLength}ä¸ªä»»åŠ¡${Array(pendingTipDotCount).fill('Â·').join('')}`))
    }
  }
  return status
}
