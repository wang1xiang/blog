  /** æ‰€æœ‰å›¾è¡¨çš„é»˜è®¤é…ç½® */
  const DEFAULT_OPTIONS: ECOption = {

      {

          show: false,

        data: props.xAxisData,
      },
    ],

      {
        type: 'value',
        minInterval: 1,

    ],

      // https://echarts.apache.org/zh/option.html#color

      backgroundColor: 'transparent',

        trigger: 'axis',
        axisPointer: {
          type: 'shadow',
        },
      },
      legend,
      grid: {
        left: 10,
        right: 10,
        bottom: props.dataZoom ? 40 : 10,
        containLabel: true,
      },
      // å³ä¸Šè§’å·¥å…·æ

        show: false,
        feature: {
          magicType: {
            type: ['line', 'bar'],
          },
          dataView: {
            readOnly: false,
          },
          saveAsImage: {},
        },
      },
      dataset: {
        source: props.datasetSource,
      },
      xAxis,
      yAxis,
      dataZoom,
      series,
    }
