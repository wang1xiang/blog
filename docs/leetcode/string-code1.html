var replaceNumber = function (s) {
  let arr = s.split('')
  const a = 'a'.charCodeAt()
  const z = 'z'.charCodeAt()
  const isStr = (str) => str.charCodeAt() >= a && str.charCodeAt() <= z

  let n = 0
  for (let i of arr) {
    n += isStr(i) ? 1 : 6
  }

  const newArr = new Array(n).fill(0)
  let index = arr.length - 1
  for (let i = n - 1; i >= 0; i--) {
    if (isStr(arr[index])) {
      newArr[i] = arr[index]
    } else {
      newArr[i] = 'r'
      newArr[i - 1] = 'e'
      newArr[i - 2] = 'b'
      newArr[i - 3] = 'm'
      newArr[i - 4] = 'u'
      newArr[i - 5] = 'n'
      i -= 5
    }
    index--
  }

  return newArr.join('')
}
