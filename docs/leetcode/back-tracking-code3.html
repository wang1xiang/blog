var findSubsequences = function (nums) {
  const result = []
  const path = []

  const backtracking = (startIndex) => {
    const set = new Set()
    if (path.length >= 2) {
      result.push([...path])
    }
    for (let i = startIndex; i < nums.length; i++) {
      const num = nums[i]
      if (set.has(num)) continue
      if (path.length > 0 && num < path[path.length - 1]) continue

      path.push(num)
      set.add(num)
      backtracking(i + 1)
      path.pop()
    }
  }
  backtracking(0)
  return result
}
