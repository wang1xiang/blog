function preorderTraversal(root) {
  const result = []
  const stack = [root]
  while (stack.length) {
    const cur = stack.pop()

    if (!cur) {

function levelOrder(root) {
  if (!root) return []
  const result = []
  const queue = [root]

  while (queue.length) {
    const size = queue.length
    const curLevel = []

    for (let i = 0; i < size; i++) {
