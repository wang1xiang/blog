var reverseStr = function (s, k) {
  let arr = s.split('')
  for (let i = 0; i < arr.length; i += 2 * k) {
    let left = i
    let right = i + k - 1 >= length - 1 ? length - 1 : i + k - 1
    while (left < right) {

    }
  }
  return arr.join('')
}
